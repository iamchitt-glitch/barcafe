ข้อมูลร้าน
ใส่ถ้ามีค่าเช่าคงที่ทุกเดือน
เชื่อมต่อ Google Sheets
วาง URL จาก Apps Script ที่ Deploy แล้ว
วิธีเชื่อม Google Sheets (ทำครั้งเดียว)
1
เปิด Google Sheets ของคุณ
เปิดไฟล์ Sheets ที่ใช้งานอยู่ แล้วไปที่เมนู Extensions → Apps Script
2
วาง Code นี้ลงใน Apps Script
ลบ Code เดิมทั้งหมดออก แล้ววาง Code ด้านล่างนี้แทน
// BAR CAFE — Google Apps Script // วางใน Extensions > Apps Script แล้ว Deploy เป็น Web App function doPost(e) { try { const body = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const type = body.type; const data = body.data; if (type === 'income') { // เขียนลงชีตรายเดือน เช่น Jan26, Feb26 const sheetName = getMonthSheet(data.date); let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow(['วันที่','ยอดขาย','จำนวนลูกค้า','ประเภทลูกค้า','หมายเหตุ']); sheet.getRange(1,1,1,5).setFontWeight('bold'); } // ตรวจสอบว่ามีวันนั้นแล้วหรือยัง const vals = sheet.getDataRange().getValues(); const rowIdx = vals.findIndex(r => r[0] === data.date); const row = [data.date, data.amount, data.customers||0, data.type||'', data.note||'']; if (rowIdx > 0) { sheet.getRange(rowIdx+1, 1, 1, 5).setValues([row]); } else { sheet.appendRow(row); } } if (type === 'expense') { const sheetName = getMonthSheet(data.date) + '_expense'; let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow(['วันที่','หมวด','รายละเอียด','จำนวนเงิน']); sheet.getRange(1,1,1,4).setFontWeight('bold'); } sheet.appendRow([data.date, data.cat, data.desc||'', data.amount]); } return ContentService .createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function getMonthSheet(dateStr) { const d = new Date(dateStr); const months = ['Jan','Feb','Mar','Apr','May','Jun', 'Jul','Aug','Sep','Oct','Nov','Dec']; return months[d.getMonth()] + String(d.getFullYear()).slice(2); }
3
Deploy เป็น Web App
กด Deploy → New deployment → เลือก Web App → Execute as: Me → Who has access: Anyone → กด Deploy → Copy URL
4
วาง URL ในช่องด้านบน
นำ URL ที่ได้มาวางในช่อง "Google Apps Script URL" แล้วกด บันทึก ระบบจะส่งข้อมูลเข้า Sheets อัตโนมัติทุกครั้งที่บันทึก
ข้อมูลในเครื่อง
บันทึก URL แล้ว — ทดสอบกรอกข้อมูลดูได้เลย
